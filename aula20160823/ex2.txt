#include <stdio.h>
#include <time.h>

int main(void)
{
    printf("\t\t\t\tPar ou impar \n\n");
    int num,soma,aleatorio;
    printf("Informe um numero inteiro nao negativo: ");
    scanf("%d",&num);
    srand(time(0));
    aleatorio = rand()%2;
    soma = num + aleatorio;

    if(soma % 2 == 0)
    {
        printf("\nNumero: %d\nNumero aleatorio entre 0 e 1: %d\nO resultado da soma e par, %d.\n\n\n",num,aleatorio,soma);
    }
    else
    {
        printf("\nNumero: %d\nNumero aleatorio entre 0 e 1: %d\nO resultado da soma e impar, %d.\n\n\n",num,aleatorio,soma);
    }

    return 0;
}
