#include <stdio.h>

int main(void)
{
    printf("\t\t\t\tNumeros perfeitos\n\n");
    int num,soma=0,aux;

    scanf("%d",&num);
    aux = num-1;

    while(aux > 0)
    {
        if(num % aux == 0)
            soma+= aux;
        aux--;
    }
    if(soma == num && num != 0)
        printf("\nO numero %d e um numero perfeito!",num);
    else
        printf("\nO numero %d nao e um tipo perfeito!\n\n",num);

    return 0;
}
