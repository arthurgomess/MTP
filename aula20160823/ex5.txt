#include <stdio.h>
#include <time.h>

int main(void)
{
    printf("\t\t\t\tJogo de adivinha \n\n");
    int num,aleatorio;

    srand(time(0));
    aleatorio = rand()%100;

    printf("Informe um numero inteiro que voce acha que e: ");
    scanf("%d",&num);

    if(num == aleatorio)
    {
        printf("Parabens voce acertou de primeira!\n\n\n");
    }
    else
    {
        printf("\n");
        while(num != aleatorio)
        {
            printf("\nVoce errou! ");
            if(num > aleatorio)
            {
                printf("Alto!\n");
            }
            else
            {
                printf("Baixo!\n");
            }
            printf("\nInforme outro numero: ");
            scanf("%d",&num);
        }
        if(num == aleatorio)
        {
            printf("Parabens voce acertou!\n\n\n");
        }
    }

    return 0;
}
