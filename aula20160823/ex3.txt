#include <stdio.h>
#include <time.h>

int main(void)
{
    printf("\t\t\t\t7 or 11\n\n");
    int soma,aleatorio1, aleatorio2, aleatorio3;
    srand(time(0));
    do
    {
        aleatorio1 = rand()%7;
    }
    while (aleatorio1 == 0);

    do
    {
        aleatorio2 = rand()%7;
    }
    while (aleatorio2 == 0);

    do
    {
        aleatorio3 = rand()%7;
    }
    while (aleatorio3 == 0);

    soma = aleatorio1 + aleatorio2 + aleatorio3;

    printf("O valor do dado1: %d\n",aleatorio1);
    printf("O valor do dado2: %d\n",aleatorio2);
    printf("O valor do dado3: %d\n",aleatorio3);
    if(soma == 7 || soma == 11)
    {
        printf("O resultado da soma e: %d. Voce ganhou!!!",soma);
    }
    else
    {
        printf("O resultado da soma e: %d. Voce perdeu!!!",soma);
    }

    return 0;
}
