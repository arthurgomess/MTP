#include <stdlib.h>
#include <stdio.h>

int main ()
{
    printf("\t\t\t\tMedia ponderada\n\n");
    float A, B, C, nota;

    printf("Insira as notas do aluno seguidas de um espaco entre cada: ");
    scanf("%f",&A);
    scanf("%f",&B);
    scanf("%f",&C);
    nota = (A*2+B*3+C*5)/(2+3+5);
    printf("\nA media ponderada do aluno e: %.2f\n\n",nota);

    return 0;
}
